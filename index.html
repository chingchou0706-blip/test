<script>

  // ====== Demo 版本：用 localStorage 存資料（上 GitHub Pages 可用）======

  const LS_KEY = "smart_expense_records_v1";

 

  function loadRecords() {

    try { return JSON.parse(localStorage.getItem(LS_KEY) || "[]"); }

    catch { return []; }

  }

  function saveRecords(records) {

    localStorage.setItem(LS_KEY, JSON.stringify(records));

  }

 

  const defaultConfig = { app_title: "智慧記帳", currency_symbol: "NT$" };

 

  let allRecords = loadRecords();

  let customCategories = [];

 

  let currentFlow = {

    category: "",

    paymentMethod: "",

    paymentColor: "",

    paymentEmoji: "",

    carrierUsed: false,

    lastDeepLink: ""

  };

 

  function updateUI() {

    updateTotalAmount();

    updateCategoryStats();

    updateRecentRecords();

  }

 

  function updateTotalAmount() {

    const total = allRecords.reduce((sum, r) => sum + (Number(r.amount) || 0), 0);

    document.getElementById('total-amount').textContent = total.toLocaleString();

  }

 

  function updateCategoryStats() {

    const statsContainer = document.getElementById('category-stats');

    const categoryTotals = {};

 

    allRecords.forEach(record => {

      categoryTotals[record.category] = (categoryTotals[record.category] || 0) + (Number(record.amount) || 0);

    });

 

    const sorted = Object.entries(categoryTotals).sort((a,b)=>b[1]-a[1]).slice(0,5);

    if (sorted.length === 0) {

      statsContainer.innerHTML = '<p class="text-gray-500 text-center py-4">尚無消費記錄</p>';

      return;

    }

 

    const maxAmount = Math.max(...sorted.map(([_, amount]) => amount));

    statsContainer.innerHTML = sorted.map(([category, amount]) => {

      const percentage = (amount / maxAmount) * 100;

      return `

        <div class="bg-white rounded-xl p-4 shadow-sm">

          <div class="flex justify-between mb-2">

            <span class="font-semibold" style="color:#2d3748;">${category}</span>

            <span class="font-bold" style="color:#667eea;">${defaultConfig.currency_symbol}${amount.toLocaleString()}</span>

          </div>

          <div class="w-full bg-gray-200 rounded-full h-2">

            <div class="h-2 rounded-full" style="width:${percentage}%; background:#667eea;"></div>

          </div>

        </div>

      `;

    }).join('');

  }

 

  function updateRecentRecords() {

    const container = document.getElementById('recent-records');

    const sorted = [...allRecords].sort((a,b)=>new Date(b.timestamp)-new Date(a.timestamp)).slice(0,10);

 

    if (sorted.length === 0) {

      container.innerHTML = '<p class="text-gray-500 text-center py-4">尚無消費記錄</p>';

      return;

    }

 

    container.innerHTML = sorted.map(record => {

      const date = new Date(record.timestamp);

      const timeStr = `${date.getMonth()+1}/${date.getDate()} ${String(date.getHours()).padStart(2,'0')}:${String(date.getMinutes()).padStart(2,'0')}`;

      return `

        <div class="bg-white rounded-xl p-4 shadow-sm flex justify-between items-center">

          <div>

            <div class="font-semibold text-lg mb-1" style="color:#2d3748;">${record.category}</div>

            <div class="text-sm text-gray-500">${record.payment_method} · ${timeStr}</div>

            ${record.note ? `<div class="text-sm text-gray-400 mt-1">${record.note}</div>` : ''}

          </div>

          <div class="text-xl font-bold" style="color:#667eea;">${defaultConfig.currency_symbol}${Number(record.amount).toLocaleString()}</div>

        </div>

      `;

    }).join('');

  }

 

  function showScreen(screenId) {

    const screens = ['main-screen','step1-screen','step2-screen','step3-screen','step4-screen','payment-redirect-screen'];

    screens.forEach(id => {

      const el = document.getElementById(id);

      if (!el) return;

      if (id === screenId) {

        el.classList.remove('hidden');

        el.classList.add('flex','fade-in');

      } else {

        el.classList.add('hidden');

        el.classList.remove('flex');

      }

    });

  }

 

  const defaultCategories = ["早餐","午餐","晚餐","交通","娛樂","購物","其他"];

 

  function getCategoryEmoji(category) {

    const map = { 早餐:'🍳', 午餐:'🍱', 晚餐:'🍽️', 交通:'🚗', 娛樂:'🎮', 購物:'🛍️', 其他:'📝' };

    return map[category] || '💰';

  }

 

  function renderCategoryGrid() {

    const grid = document.getElementById('category-grid');

    const allCategories = [...defaultCategories, ...customCategories];

 

    grid.innerHTML = allCategories.map(c => `

      <button class="category-btn p-6 rounded-2xl bg-white shadow-sm text-center font-semibold transition-all hover:shadow-md"

              data-category="${c}" style="color:#2d3748;">

        ${getCategoryEmoji(c)}<br><span class="text-sm">${c}</span>

      </button>

    `).join('');

 

    document.querySelectorAll('.category-btn').forEach(btn => {

      btn.addEventListener('click', () => {

        currentFlow.category = btn.dataset.category;

        document.getElementById('selected-category-display').textContent = `分類: ${currentFlow.category}`;

        document.getElementById('selected-category-display2').textContent = `分類: ${currentFlow.category}`;

  // toast

  function toast(msg) {

    const div = document.createElement('div');

    div.textContent = msg;

    div.className = 'fixed top-20 left-1/2 transform -translate-x-1/2 bg-black bg-opacity-80 text-white px-5 py-3 rounded-xl shadow-lg z-50 text-sm';

    document.body.appendChild(div);

    setTimeout(() => div.remove(), 2200);

  // iOS Deep Link（demo：嘗試開啟支付功能/付款碼頁）

  const iosDeepLinks = {

    'LINE Pay': 'line://pay/generateQR',

    'Taiwan Pay': 'twmpshortcut://?type=payment',

    '全支付': 'com.pxpay.plus://zjdja',

    '街口支付': 'jkos://showQRCode',

    '悠遊付': 'tw.com.easycard.easycardwallet://paymentCode'

  };

 

  function isIOS() {

    return /iPad|iPhone|iPod/.test(navigator.userAgent) && !window.MSStream;

  }

 

  function startIOSDeepLinkJump(method) {

    if (!isIOS()) {

      toast('此 Demo 目前以 iOS Safari 實測為主');

      return;

    }

 

    const url = iosDeepLinks[method];

    if (!url) {

      toast('此支付方式尚未設定 iOS Deep Link');

      return;

    }

 

    currentFlow.lastDeepLink = url;

 

    document.getElementById('redirect-payment-name').textContent = currentFlow.paymentMethod;

    document.getElementById('redirect-payment-icon').textContent = currentFlow.paymentEmoji;

    document.getElementById('redirect-payment-icon').style.background = currentFlow.paymentColor;

    document.getElementById('pulse-ring').style.background = currentFlow.paymentColor + '40';

    document.getElementById('deep-link-url').textContent = url;

 

    showScreen('payment-redirect-screen');

 

    const start = Date.now();

    setTimeout(() => { window.location.href = url; }, 0);

 

    setTimeout(() => {

      if (Date.now() - start < 2000) {

        toast(`無法開啟 ${method}（可能未安裝或路徑不支援）`);

        showScreen('step3-screen');

      }

    }, 1400);

  }
